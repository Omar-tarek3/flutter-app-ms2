void main() => runApp(MaterialApp(

class Home extends StatelessWidget {

    return Scaffold(
    appBar: AppBar(
    title: Text('my app'),
    ) ,
    body: Column(
    children: [
    TextFormField(
    decoration: const InputDecoration(
    hintText: 'user name',
    ),
    ),
    TextFormField(
    decoration: const InputDecoration(
    hintText: 'password',

  }
}
